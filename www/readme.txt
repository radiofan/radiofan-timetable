Описание куки
sid - string - токен сессии
tmt => array(
	'p' => array( - список разделов
		array(
			'i' => int - id
			't' => int - тип раздела (0:группа, 1:препод, 2:кабинет)
			'n' => string - имя раздела, длина MAX_PART_NAME_LEN
			'o' => string - порядок столбцов
		)
		...
	),
	'o' => array( - список опций
		'tah' => bool - скрывать звание препода
		'wwr' => bool - перенос слов
		'gcd' => int - идти к текущему (0:нет, 1:неделе, 2:дню)
		'lun' => bool - объединять уроки
	),
	's' => array( - размеры столбцов
		'nsz' => int - размер столбца с номером пары
		'tsz' => int - размер столбца со временем пары
		'p' => array( - массив разделов
			array( - размеры столбцов разделов
				'l' - int размер столбца урок
				't' - int размер столбца тип
				'g' - int размер столбца группа
				'c' - int размер столбца кабинет
				'p' - int размер столбца препод
			)
		)
	)
)
